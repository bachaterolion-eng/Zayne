<script>
    const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    const notes = {
        'C': 261.63, 'C#': 277.18, 'D': 293.66, 'D#': 311.13,
        'E': 329.63, 'F': 349.23, 'F#': 369.99, 'G': 392.00,
        'G#': 415.30, 'A': 440.00, 'A#': 466.16, 'B': 493.88
    };
    const noteKeys = Object.keys(notes);
    
    let currentRound = 1;
    let targetNote = '';
    let currentPool = [];

    function playTone(freq) {
        const osc = audioCtx.createOscillator();
        const gain = audioCtx.createGain();
        osc.type = 'triangle';
        osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
        gain.gain.setValueAtTime(0.5, audioCtx.currentTime);
        gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 1);
        osc.connect(gain);
        gain.connect(audioCtx.destination);
        osc.start();
        osc.stop(audioCtx.currentTime + 1);
    }

    function initiatePreview() {
        currentRound = parseInt(document.getElementById('start-round').value) || 1;
        document.getElementById('round-display').innerText = currentRound;
        document.getElementById('preview-round-num').innerText = currentRound;
        
        // Determine how many notes are in this round (at least 2, max 12)
        const poolSize = Math.min(2 + Math.floor(currentRound / 2), 12);
        currentPool = noteKeys.slice(0, poolSize);
        
        // Show preview UI
        document.getElementById('preview-section').classList.remove('hidden');
        document.getElementById('main-controls').classList.add('hidden');
        document.getElementById('message').innerText = "Testing Phase...";
        renderButtons(true);
    }

    async function playCurrentPool() {
        for (let note of currentPool) {
            playTone(notes[note]);
            document.getElementById('message').innerText = "Listening to: " + note;
            await new Promise(r => setTimeout(r, 800));
        }
        document.getElementById('message').innerText = "Preview complete!";
    }

    function startActualRound() {
        document.getElementById('preview-section').classList.add('hidden');
        document.getElementById('main-controls').classList.remove('hidden');
        renderButtons(false);
        nextSequence();
    }

    function renderButtons(disabled) {
        const grid = document.getElementById('note-buttons');
        grid.innerHTML = '';
        currentPool.forEach(note => {
            const btn = document.createElement('button');
            btn.className = 'btn-note';
            btn.innerText = note;
            btn.onclick = () => checkAnswer(note, btn);
            if(disabled) btn.disabled = true;
            grid.appendChild(btn);
        });
    }

    function nextSequence() {
        targetNote = currentPool[Math.floor(Math.random() * currentPool.length)];
        playTone(notes[targetNote]);
        document.getElementById('message').innerText = "Identify the note...";
    }

    function checkAnswer(note, btn) {
        if (note === targetNote) {
            btn.classList.add('correct');
            document.getElementById('message').innerText = "Correct!";
            setTimeout(() => {
                btn.classList.remove('correct');
                currentRound++;
                document.getElementById('start-round').value = currentRound;
                initiatePreview(); // Go back to preview for the next round
            }, 1000);
        } else {
            btn.classList.add('wrong');
            document.getElementById('message').innerText = "Try again!";
            setTimeout(() => btn.classList.remove('wrong'), 500);
            playTone(notes[targetNote]); // Replay target
        }
    }
</script>
